namespace aether::resources {

// 资源加载器 — 公开接口
export class ResourceLoader {
public:
    ResourceLoader(std::shared_ptr<rhi::Device> device, core::ThreadPool& pool);

    // 异步加载 Mesh，返回 MeshResource（BufferPtr 顶点/索引）
    core::Task<MeshResource> LoadMeshAsync(std::string_view path);

    // 异步加载纹理
    core::Task<TextureResource> LoadTextureAsync(std::string_view path);

    // 异步加载材质描述
    core::Task<MaterialResource> LoadMaterialAsync(std::string_view path);

private:
    core::Task<MeshResource> LoadMeshInternal(std::string path);
    core::Task<TextureResource> LoadTextureInternal(std::string path);

    std::shared_ptr<rhi::Device> m_device;
    core::ThreadPool& m_threadPool;
};

// 资源类型 (导出)
export struct MeshResource {
    rhi::BufferPtr vertexBuffer;
    rhi::BufferPtr indexBuffer;
    uint32_t vertexCount;
    uint32_t indexCount;
    // LOD 级别数据
    std::vector<MeshLOD> lods;
};

export struct TextureResource {
    rhi::TexturePtr texture;
    uint32_t width, height;
    rhi::Format format;
};

export struct MaterialResource {
    // 材质参数，后续传入 ShaderBinding
    MaterialParams params;
    std::vector<TextureResource> textures;
};

} // namespace aether::resources

协程加载流程示例

core::Task<MeshResource> ResourceLoader::LoadMeshAsync(std::string_view path) {
    // 1. 在后台线程读取文件（不阻塞主线程）
    std::vector<std::byte> fileData = co_await core::ReadFileAsync(path);

    // 2. 在后台线程解码 (Assimp 或 gltf 解析)
    MeshData decoded = co_await core::RunOnThreadPool([&fileData]() {
        return DecodeMesh(fileData);
    });

    // 3. 回到 RHI 线程创建 GPU 缓冲
    auto vb = m_device->CreateBuffer(BufferDesc::Vertex(decoded.vertices));
    auto ib = m_device->CreateBuffer(BufferDesc::Index(decoded.indices));

    // 4. 通过 Copy Queue 上传到 GPU
    auto copyQueue = m_device->GetCopyQueue();
    copyQueue->Upload(vb, decoded.vertices.data());
    copyQueue->Upload(ib, decoded.indices.data());
    m_device->ExecuteCopyCommands(std::move(copyQueue));

    co_return MeshResource{
        .vertexBuffer = std::move(vb),
        .indexBuffer  = std::move(ib),
        .vertexCount  = decoded.vertexCount,
        .indexCount   = decoded.indexCount,
        .lods         = std::move(decoded.lods)
    };
}

使用示例（Renderer 中调用）

class Scene {
    ResourceLoader m_loader;

    core::Task<void> LoadSceneAsync() {
        // 并发加载多个资源
        auto meshTask  = m_loader.LoadMeshAsync("models/character.gltf");
        auto texTask   = m_loader.LoadTextureAsync("textures/character.png");

        // 等待两者都完成
        auto [mesh, tex] = co_await core::WhenAll(std::move(meshTask), std::move(texTask));

        // 创建 GPU 场景数据
        auto obj = SceneObject{};
        obj.meshIndex = m_gpuSceneData.AddMesh(mesh);
        obj.materialIndex = m_gpuSceneData.AddMaterial(tex);

        co_return;
    }
};
