CommandList 核心设计

所有 GPU 操作通过 CommandList 记录，提交到 GPU 执行。这是 D3D12 和 Vulkan 的共性抽象：

// 伪代码：CommandList 接口设计
export interface CommandList {
    void Reset();
    void Close();

    // 通用
    void ResourceBarrier(BarrierDesc);
    void Clear(Buffer/Texture, ...);
    void Copy(Resource src, Resource dst);
    void BindPipeline(PipelineState);
    void BindDescriptor(ShaderBinding, uint slot);
    void PushConstants(uint size, void* data);
}

export interface GraphicsCommandList : CommandList {
    void SetViewport(Viewport);
    void SetScissor(Rect);
    void IASetVertexBuffer(Slot, Buffer, Stride);
    void IASetIndexBuffer(Buffer, Format);
    void Draw(uint vertexCount, uint instanceCount);
    void DrawIndexed(uint indexCount, uint instanceCount);
    void DrawIndirect(Buffer args, uint offset);        // ← GPU Driven 关键
    void DispatchRays(RayTracingDesc);                   // ← Ray Tracing 关键
    void DispatchMesh(uint threadGroupCount);             // ← Mesh Shader 关键
}

export interface ComputeCommandList : CommandList {
    void Dispatch(uint groupX, uint groupY, uint groupZ);
    void DispatchIndirect(Buffer args, uint offset);
}

后端实现策略

D3D12 实现

  • 直接映射到 ID3D12GraphicsCommandList6
  • 使用 ID3D12Device10 + D3D12_FEATURE 做功能检测
  • ShaderBinding → Descriptor Heap + Table
  • Mesh Shader → ID3D12GraphicsCommandList::DispatchMesh
  • Ray Tracing → ID3D12GraphicsCommandList5::DispatchRays + StateObject

Vulkan 实现

  • 直接映射到 VkCommandBuffer
  • 使用 vkGetPhysicalDeviceFeatures2 检测支持
  • ShaderBinding → Descriptor Set
  • Mesh Shader → VK_NV_mesh_shader / VK_EXT_mesh_shader
  • Ray Tracing → VK_KHR_acceleration_structure + VK_KHR_ray_tracing_pipeline

资源生命周期

所有 GPU 资源通过 Device 创建，返回句柄。资源由引用计数管理：

// Device 是唯一创建资源的入口
export interface Device {
    Buffer*    CreateBuffer(const BufferDesc&);
    Texture*   CreateTexture(const TextureDesc&);
    PipelineState* CreateGraphicsPipeline(const GfxPipelineDesc&);
    PipelineState* CreateRayTracingPipeline(const RTPipelineDesc&);
    ShaderBinding* CreateShaderBinding(const BindingLayout&);

    // 资源上传
    CommandList* GetCopyQueue(); // 异步上传用
    void         ExecuteCopyCommands(CommandList*);

    // 同步
    Fence*       CreateFence();
    void         WaitForFence(Fence*, uint64 value);
}
